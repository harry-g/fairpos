import com.openbravo.pos.payment.PaymentInfo;
import javax.swing.JOptionPane;

boolean isCash = false;
String change = "";
for (i=0; i<ticket.payments.size(); i++) {
    PaymentInfo p = ticket.payments.get(i);

    if ("cash".equals(p.getName())) {
        isCash = true;
        change = p.printChange();
    }
}
if(isCash) {
    JOptionPane.showMessageDialog(null, "Bitte Rückgeld " + change + " geben.", "Wechselgeld", JOptionPane.PLAIN_MESSAGE);
}

sales.printerStart();
